### SETUP
# Parameters
# h_ are human units (um, mm/s, etc, usually floats), m_ are machine units (all ints). Ideally h_ variables will be writable registers
h_pulse_dist_m1: float = 1.25           # um, float, x8 microstep distance. Could be different for motors
h_speed_limit_m1: int = 100             # mm/s or nm/us, human units, u8. Could be a float
h_speed_start_m1: float = 1.0           # mm/s, starting speed, float. Could be int, but want margin
h_acc_m1: float = 1.0                   # m/s2, human units, float. Likely no quicker than this.
h_move_dist_m1: float = 15.0            # mm, human units, how far for fast move, float
forward_direction_m1: int = 1           # 1 = forward, 0 = backward, u8. I don't think this is necessary if h_move_dist_m1 has a sign


# Parameter Conversion
# All floats in Parameters should be converted to integers in machine units
m_speed_start_m1: int = int(1000*h_speed_start_m1/h_pulse_dist_m1) # machine units, upulse/us, u32 
m_speed_limit_m1: int = int(1000*h_speed_limit_m1/h_pulse_dist_m1) # machine units, upulse/us, u32 
m_acc_m1: int = int(1000*h_acc_m1/h_pulse_dist_m1)  # u16 upulse/us2 
move_pulses_m1: int = int(h_move_dist_m1*1000/h_pulse_dist_m1)  # pulses to move, u16


# Variables
m_speed_m1: int = m_speed_start_m1      # Start at start speed, u32
#position_m1: int = 0                   # um u16
delay_m1: int = 0                       # us, starting value. u32 solely for calculation precision
accelerating_m1: int = 1                # 0 is decelerating, could be u8 or boolean
pulses_to_accelerate_m1: int = 0         # counter to know when to decelerate
short_move_pulses_m1: int = move_pulses_m1//2     # Use to compare for short moves that don't hit speed limit

#Set up starting point
#### ONLY FOR PLOTS
x = [0]                                 #mm, need 0 for starting point on plot
t = [0]                                 #μs, need 0 for starting point on plot
d = []                                  #μs
v = []                                  #mm/s




### LOOP
print('Pulses: ', move_pulses_m1)       # For reference for plots
# Delay is same for accelerating or decelerating, forward or backwards
for i in range(0,move_pulses_m1):            # ONLY FOR PLOTS TO RUN LOOP
    
    delay_m1 = (1000000)//(m_speed_m1)       # us. pulse/speed, 1000000 converts units. Floor divide so only whole us 
    move_pulses_m1 = move_pulses_m1 - 1      # decrement pulses to move
    
    

    ## Can send or queue Pulse here, and compute the next pulse below during the delay. Use forward_direction_m1 here if used.
    # Speed has checks
    if accelerating_m1 == 1:
        if m_speed_m1 < m_speed_limit_m1:
            m_speed_m1 = m_speed_m1 + ((m_acc_m1*delay_m1)//1000)     # floor divide //1000 for unit conversion
            #print('v', m_speed_m1)
            pulses_to_accelerate_m1 = pulses_to_accelerate_m1 + 1     # increment
        else:
            speed_m1 = m_speed_limit_m1

    else:   #Decelerating
        if m_speed_m1 > m_speed_start_m1:
            m_speed_m1 = m_speed_m1 - ((m_acc_m1*delay_m1)//1000)     # floor divide //1000 for unit conversion
        else:
            m_speed_m1 = m_speed_start_m1     # Shouldn't be necessary?
    if accelerating_m1 == 1:
        if move_pulses_m1 == pulses_to_accelerate_m1 or short_move_pulses_m1 == pulses_to_accelerate_m1:
            accelerating_m1 = 0                 # start decelerating
            print('hit deceleration')           # Just a check

    d.append(delay_m1)
    t.append(t[-1]+delay_m1)
    x.append(x[-1]+(h_pulse_dist_m1/1000))
    v.append(m_speed_m1*h_pulse_dist_m1/1000)

x.pop(0)  #Just temp values for first speed and position values
t.pop(0)
